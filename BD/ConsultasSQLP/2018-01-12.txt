create database biblioteca2018
use biblioteca2018
create table autor (
   nombre varchar(30) primary key,
   fecnac date
) 
create table libro (
   isbn varchar(22) primary key,
   titulo varchar(50)
)
create table escribe (
   nombreA varchar(30) references autor,
   idL varchar(22) references libro,
   constraint PK_nombreA_idL 
       primary key (nombreA, idL)
)
create table ejemplar (
   idL varchar(22) references libro
      on delete cascade
      on update cascade,
   nEjemplar int,
   editorial varchar(40),
   edicion int,
   estado varchar(20) 
      check (estado in ('regular',
                         'malo', 'bueno')),
   primary key(idL, nEjemplar)                         
)
create table socio (
   idSocio varchar(40) primary key,
   direccion varchar(50)
)
create table presta (
   idS varchar(40) references socio,
   idL varchar(22),
   nEjem int,
   constraint FK_idL_nEjem 
      foreign key(idL, nEjem)
      references ejemplar
)

select nombreA, COUNT(*) as 'Número de libros',
     (select fecnac
        from autor
        where nombre=nombreA) as 'Fecha de nacimiento'
    from escribe
    group by nombreA

select nombre, COUNT(idL) as 'Número de libros',
  min(fecnac)
    from escribe right join autor
     on nombreA = nombre
    group by nombre
